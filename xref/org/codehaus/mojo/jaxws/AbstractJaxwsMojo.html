237     protected List<String> getCommonArgs() throws MojoExecutionException {
238         List<String> commonArgs = new ArrayList<String>();
239 
240         if (!isDefaultSrc(getSourceDestDir()) || keep) {
241             commonArgs.add("-keep");
242             commonArgs.add("-s");
243             commonArgs.add("'" + getSourceDestDir().getAbsolutePath() + "'");
244             if (!getSourceDestDir().mkdirs() && !getSourceDestDir().exists()) {
245                 getLog().warn("Cannot create directory: " + getSourceDestDir().getAbsolutePath());
246             }
247             addSourceRoot(getSourceDestDir().getAbsolutePath());
248         }
249 
250         File destDir = getDestDir();
251         if (!destDir.mkdirs() && !destDir.exists()) {
252             getLog().warn("Cannot create directory: " + destDir.getAbsolutePath());
253         }
254         commonArgs.add("-d");
255         commonArgs.add("'" + destDir.getAbsolutePath() + "'");
256 
257         if (verbose) {
258             commonArgs.add("-verbose");
259         }
260 
261         if (isArgSupported("-encoding")) {
262             if (encoding != null) {
263                 maybeUnsupportedOption("-encoding", encoding, commonArgs);
264             } else {
265                 getLog().warn("Using platform encoding (" + System.getProperty("file.encoding") + "), build is platform dependent!");
266             }
267         }
268 
269         if (isExtensionOn()) {
270             commonArgs.add("-extension");
271         }
272 
273         if(getXnocompile()){
274             commonArgs.add("-Xnocompile");
275         }
276 
277         // add additional command line options
278         if (args != null) {
279             for (String arg : args) {
280                 commonArgs.add(arg);
281             }
282         }
283         return commonArgs;
284     }

286     protected boolean isArgSupported(String arg) throws MojoExecutionException {
287         //try Metro first
288         Artifact a = pluginDescriptor.getArtifactMap().get("org.glassfish.metro:webservices-tools");
289         List<String> supportedArgs = null;
290         String v = null;
291         try {
292             if (a != null) {
293                 ArtifactVersion av = a.getSelectedVersion();
294                 v = av.toString();
295                 if (av.getMajorVersion() == 2 && av.getMinorVersion() == 2 && av.getIncrementalVersion() == 0) {
296                     supportedArgs = METRO_22;
297                 } else if (av.getMajorVersion() == 2 && av.getMinorVersion() == 2 && av.getIncrementalVersion() >= 1) {
298                     supportedArgs = METRO_221;
299                 } else { //if (av.getMajorVersion() >= 2 && av.getMinorVersion() >= 3) {
300                     supportedArgs = METRO_23;
301                 }
302             } else {
303                 //fallback to RI
304                 a = pluginDescriptor.getArtifactMap().get("com.sun.xml.ws:jaxws-tools");
305                 ArtifactVersion av = a.getSelectedVersion();
306                 v = av.toString();
307                 if (av.getMajorVersion() == 2 && av.getMinorVersion() == 2 && av.getIncrementalVersion() == 6) {
308                     supportedArgs = METRO_22;
309                 } else if (av.getMajorVersion() == 2 && av.getMinorVersion() == 2 && av.getIncrementalVersion() == 7) {
310                     supportedArgs = METRO_221;
311                 } else { //if (av.getMajorVersion() >= 2 && av.getMinorVersion() >= 2 && av.getIncrementalVersion() >= 8) {
312                     supportedArgs = METRO_23;
313                 }
314             }
315         } catch (OverConstrainedVersionException ex) {
316             throw new MojoExecutionException(ex.getMessage(), ex);
317         }
318         boolean isSupported = supportedArgs.contains(arg);
319         if (!isSupported) {
320             getLog().warn("'" + arg + "' is not supported by " + a.getArtifactId() + ":" + v);
321         }
322         return isSupported;
323     }

329     protected void exec(List<String> args) throws MojoExecutionException {
330         StreamConsumer sc = new DefaultConsumer();
331         try {
332             Commandline cmd = new Commandline();
333             if (executable != null) {
334                 if (executable.isFile() && executable.canExecute()) {
335                     cmd.setExecutable(executable.getAbsolutePath());
336                     if (getExtraClasspath() !=  null) {
337                         cmd.createArg().setLine("-cp");
338                         cmd.createArg().setValue(getExtraClasspath());
339                     }
340                 } else {
341                     throw new MojoExecutionException("Cannot execute: " + executable.getAbsolutePath());
342                 }
343             } else {
344                 cmd.setExecutable(new File(new File(System.getProperty("java.home"), "bin"), getJavaExec()).getAbsolutePath());
345                 // add additional JVM options
346                 if (vmArgs != null) {
347                     for (String arg : vmArgs) {
348                         cmd.createArg().setLine(arg);
349                     }
350                 }
351                 String[] classpath = getCP();
352                 cmd.createArg().setValue("-Xbootclasspath/p:" + classpath[0]);
353                 cmd.createArg().setValue("-cp");
354                 cmd.createArg().setValue(classpath[2]);
355                 cmd.createArg().setLine("org.codehaus.mojo.jaxws.Invoker");
356                 cmd.createArg().setLine(getMain());
357                 String extraCp = getExtraClasspath();
358                 String cp = extraCp != null ? extraCp + File.pathSeparator : "";
359                 cp += classpath[1];
360                 try {
361                     File pathFile = createPathFile(cp);
362                     cmd.createArg().setLine("-pathfile " + pathFile.getAbsolutePath());
363                 } catch (IOException ioe) {
364                     //creation of temporary file can fail, in such case just put everything on cp
365                     cmd.createArg().setValue("-cp");
366                     cmd.createArg().setValue(cp);
367                 }
368             }
369             cmd.setWorkingDirectory(project.getBasedir());
370             for (String arg : args) {
371                 cmd.createArg().setLine(arg);
372             }
373             String fullCommand = cmd.toString();
374             if (isWindows() && 8191 <= fullCommand.length()) {
375                 getLog().warn("Length of the command is limitted to 8191 characters but it has "
376                         + fullCommand.length() + " characters.");
377                 getLog().warn(fullCommand);
378             } else {
379                 getLog().debug(fullCommand);
380             }
381             if (CommandLineUtils.executeCommandLine(cmd, sc, sc) != 0) {
382                 throw new MojoExecutionException("Mojo failed - check output");
383             }
384         } catch (DependencyResolutionException dre) {
385             throw new MojoExecutionException(dre.getMessage(), dre);
386         } catch (CommandLineException t) {
387             throw new MojoExecutionException(t.getMessage(), t);
388         }
389     }

411     private String[] getCP() throws DependencyResolutionException {
412         Set<org.eclipse.aether.artifact.Artifact> endorsedCp = new HashSet<org.eclipse.aether.artifact.Artifact>();
413         Map<String, org.eclipse.aether.artifact.Artifact> cp = new HashMap<String, org.eclipse.aether.artifact.Artifact>();
414         Plugin p = pluginDescriptor.getPlugin();
415         boolean toolsFound = false;
416         for (Dependency d : p.getDependencies()) {
417             DependencyResult result = DependencyResolver.resolve(d,
418                     new ExclusionFilter(d.getExclusions()),
419                     pluginRepos, repoSystem, repoSession);
420             sortArtifacts(result, cp, endorsedCp);
421             if (containsTools(cp.keySet())) {
422                 toolsFound = true;
423             }
424         }
425         for (String dep : getExtraDependencies()) {
426             DependencyResult result = DependencyResolver.resolve(
427                     pluginDescriptor.getArtifactMap().get(dep),
428                     toolsFound ? new DepFilter(getExtraArtifactIDs()) : null,
429                     pluginRepos, repoSystem, repoSession);
430             sortArtifacts(result, cp, endorsedCp);
431         }
432         if (!containsTools(cp.keySet())) {
433             DependencyResult result = DependencyResolver.resolve(
434                     pluginDescriptor.getArtifactMap().get("com.sun.xml.ws:jaxws-tools"),
435                     null, pluginRepos, repoSystem, repoSession);
436             sortArtifacts(result, cp, endorsedCp);
437         }
438         StringBuilder sb = getCPasString(cp.values());
439         StringBuilder esb = getCPasString(endorsedCp);
440         //add custom invoker
441         String invokerPath = AbstractJaxwsMojo.class.getProtectionDomain().getCodeSource().getLocation().toExternalForm();
442         try {
443             invokerPath = new URI(invokerPath.substring(5)).getPath();
444             sb.append(invokerPath);
445         } catch (URISyntaxException ex) {
446             throw new RuntimeException(ex);
447         }
448         sb.append(File.pathSeparator);
449         //don't forget tools.jar
450         File toolsJar = new File(System.getProperty("java.home"), "../lib/tools.jar");
451         if (!toolsJar.exists()) {
452             toolsJar = new File(System.getProperty("java.home"), "lib/tools.jar");
453         }
454         sb.append(toolsJar.getAbsolutePath());
455         sb.append(File.pathSeparator);
456         getLog().debug("getCP esb: " + esb);
457         getLog().debug("getCP sb: " + sb);
458         return new String[]{esb.substring(0, ((esb.length() > 0) ? esb.length() - 1 : 0)), sb.substring(0, sb.length() - 1), invokerPath};
459     }

465     private File createPathFile(String cp) throws IOException {
466         File f = File.createTempFile("jax-ws-mvn-plugin-cp", ".txt");
467         if (f.exists() && f.isFile()) {
468             if (!f.delete()) {
469                 //this should not happen
470                 getLog().warn("cannot remove obsolete classpath setting file: " + f.getAbsolutePath());
471             }
472         }
473         Properties p = new Properties();
474         p.put("cp", cp.replace(File.separatorChar, '/'));
475         getLog().debug("stored classpath: " + cp.replace(File.separatorChar, '/'));
476         FileOutputStream fos = null;
477         try {
478             fos = new FileOutputStream(f);
479             p.store(fos, null);
480         } catch (IOException ex) {
481             logger.log(Level.SEVERE, null, ex);
482         } finally {
483             if (fos != null) {
484                 try {
485                     fos.close();
486                 } catch (IOException ex) {
487                     logger.log(Level.SEVERE, null, ex);
488                 }
489             }
490         }
491         return f;
492     }

498     private StringBuilder getCPasString(Collection<org.eclipse.aether.artifact.Artifact> artifacts) {

507     private void sortArtifacts(DependencyResult result, Map<String, org.eclipse.aether.artifact.Artifact> cp, Set<org.eclipse.aether.artifact.Artifact> endorsedCp) {
508         PreorderNodeListGenerator nlg = new PreorderNodeListGenerator();
509         FilteringDependencyVisitor visitor = new FilteringDependencyVisitor(
510                 nlg, new NotDependencyFilter(new EndorsedFilter()));
511         result.getRoot().accept(visitor);
512         for (org.eclipse.aether.artifact.Artifact a : nlg.getArtifacts(false)) {
513             cp.put(a.getGroupId() + ":" + a.getArtifactId(), a);
514         }
515 
516         nlg = new PreorderNodeListGenerator();
517         visitor = new FilteringDependencyVisitor(
518                 nlg, new EndorsedFilter());
519         result.getRoot().accept(visitor);
520         endorsedCp.addAll(nlg.getArtifacts(false));
521     }

523     private boolean containsTools(Set<String> cp) {
524         return  cp.contains("com.sun.xml.ws:jaxws-tools")
525                 || cp.contains("org.glassfish.metro:webservices-tools")
526                 || cp.contains("com.oracle.weblogic:weblogic-server-pom");

529     private static class DepFilter implements DependencyFilter {
530 
531         private final Set<Dep> toExclude = new HashSet<Dep>();
532 
533         public DepFilter(String[] artifacts) {
534             if (artifacts != null) {
535                 for (String a : artifacts) {
536                     int i = a.indexOf(':');
537                     toExclude.add(new Dep(a.substring(0, i), a.substring(i + 1)));
538                 }
539             }
540         }
541 
542         @Override
543         public boolean accept(DependencyNode node, List<DependencyNode> parents) {
544             org.eclipse.aether.artifact.Artifact a = node.getDependency().getArtifact();
545             return !toExclude.contains(new Dep(a.getGroupId(), a.getArtifactId()));
546         }
547 
548         private static class Dep {
549 
550             private final String groupId;
551             private final String artifactId;
552 
553             public Dep(String groupId, String artifactId) {
554                 this.groupId = groupId;
555                 this.artifactId = artifactId;
556             }
557 
558             @Override
559             public int hashCode() {
560                 int hash = 5;
561                 hash = 37 * hash + (this.groupId != null ? this.groupId.hashCode() : 0);
562                 return hash;
563             }
564 
565             @Override
566             public boolean equals(Object obj) {
567                 if (obj == null) {
568                     return false;
569                 }
570                 if (getClass() != obj.getClass()) {
571                     return false;
572                 }
573                 final Dep other = (Dep) obj;
574                 if ((this.groupId == null)
575                         ? (other.groupId != null)
576                         : !this.groupId.equals(other.groupId)) {
577                     return false;
578                 }
579                 //startsWith here is intentional
580                 return !((this.artifactId == null)
581                         ? (other.artifactId != null)
582                         : !this.artifactId.startsWith(other.artifactId));
583             }
584         }
